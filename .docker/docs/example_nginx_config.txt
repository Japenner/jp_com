upstream rui {
  server 127.0.0.1:4000;
}

upstream rapi {
  server 127.0.0.1:3001;
}

map $http_user_agent $limit_bots {
     default 0;
     ~*(google|bing|yandex|msnbot) 1;
     ~*(AltaVista|Googlebot|Slurp|BlackWidow|Bot|ChinaClaw|Custo|DISCo|Download|Demon|eCatch|EirGrabber|EmailSiphon|EmailWolf|SuperHTTP|Surfbot|WebWhacker) 1;
     ~*(Express|WebPictures|ExtractorPro|EyeNetIE|FlashGet|GetRight|GetWeb!|Go!Zilla|Go-Ahead-Got-It|GrabNet|Grafula|HMView|Go!Zilla|Go-Ahead-Got-It) 1;
     ~*(rafula|HMView|HTTrack|Stripper|Sucker|Indy|InterGET|Ninja|JetCar|Spider|larbin|LeechFTP|Downloader|tool|Navroad|NearSite|NetAnts|tAkeOut|WWWOFFLE) 1;
     ~*(GrabNet|NetSpider|Vampire|NetZIP|Octopus|Offline|PageGrabber|Foto|pavuk|pcBrowser|RealDownload|ReGet|SiteSnagger|SmartDownload|SuperBot|WebSpider) 1;
     ~*(Teleport|VoidEYE|Collector|WebAuto|WebCopier|WebFetch|WebGo|WebLeacher|WebReaper|WebSauger|eXtractor|Quester|WebStripper|WebZIP|Wget|Widow|Zeus) 1;
     ~*(Twengabot|htmlparser|libwww|Python|perl|urllib|scan|Curl|email|PycURL|Pyth|PyQ|WebCollector|WebCopy|webcraw) 1;
}

# Prevent long-waiting connections from wasting resources.
keepalive_requests 30;
send_timeout 30s;

limit_req_zone $binary_remote_addr zone=one:10m rate=10r/s;
limit_conn_zone $binary_remote_addr zone=addr:10m;

server {
  server_name jacobpenner.com www.jacobpenner.com;
  listen       80;
  listen  [::]:80;

  # Deny connections from bots/attackers
  if ($limit_bots = 1) {
      return 403;
  }

  # Prevent long-waiting connections from wasting resources.
  client_body_buffer_size 16K;
  client_header_buffer_size 1k;
  client_max_body_size 8m;
  large_client_header_buffers 4 8k;
  client_body_in_single_buffer on;

  # Terminate slow connections
  client_body_timeout 12s;
  client_header_timeout 12s;

  # Don't surface server tokens to the world on error
  server_tokens   off;

  #access_log  /var/log/nginx/host.access.log  main;

  add_header X-Frame-Options SAMEORIGIN;
  add_header X-Content-Type-Options nosniff;
  add_header Content-Security-Policy "default-src 'self';";
  add_header X-Permitted-Cross-Domain-Policies master-only;
  add_header Referrer-Policy same-origin;
  add_header Strict-Transport-Security 'max-age=31536000; includeSubDomains; preload';

  return 301 https://jacobpenner.com$request_uri;
}

server {
  server_name www.jacobpenner.com;

  listen 443 ssl; # managed by Certbot
  ssl_certificate /etc/letsencrypt/live/jacobpenner.com/fullchain.pem; # managed by Certbot
  ssl_certificate_key /etc/letsencrypt/live/jacobpenner.com/privkey.pem; # managed by Certbot
  include /etc/letsencrypt/options-ssl-nginx.conf; # managed by Certbot
  ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem; # managed by Certbot

  return 301 https://jacobpenner.com$request_uri;
}

server {
  server_name jacobpenner.com;

  location / {
    add_header 'Access-Control-Allow-Origin' "$http_origin";

    proxy_pass         http://rui;
    proxy_redirect     off;

    proxy_set_header   Host $host;
    proxy_set_header   X-Real-IP $remote_addr;
    proxy_set_header   X-Forwarded-For $proxy_add_x_forwarded_for;
    proxy_set_header   X-Forwarded-Host $server_name;
    proxy_set_header   X-Forwarded-Proto $scheme;

    # Limit the rate of accepting incoming requests
    limit_req zone=one burst=5;

    # Limit the number of connections that can be opened by a unique IP address.
    limit_conn addr 10;
  }

  location /api {
    # rewrite /api/(.*) /$1 break;

    proxy_pass http://rapi/api/v1;

    proxy_set_header   Host $host;
    proxy_set_header   X-Real-IP $remote_addr;
    proxy_set_header   X-Forwarded-For $remote_addr;
    proxy_set_header   X-Forwarded-Host $server_name;

    # Limit the rate of accepting incoming requests
    limit_req zone=one burst=5;

    # Limit the number of connections that can be opened by a unique IP address.
    limit_conn addr 10;
  }

  listen 443 ssl; # managed by Certbot
  ssl_certificate /etc/letsencrypt/live/jacobpenner.com/fullchain.pem; # managed by Certbot
  ssl_certificate_key /etc/letsencrypt/live/jacobpenner.com/privkey.pem; # managed by Certbot
  include /etc/letsencrypt/options-ssl-nginx.conf; # managed by Certbot
  ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem; # managed by Certbot
}
